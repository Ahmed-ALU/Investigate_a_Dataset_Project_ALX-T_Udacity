#---Cleaning the hyprid data---
# in order to do that, we are going to create a python function to make it easy.

# used some help from the classroom
def fix_hyprid(base_column, split_columns, df) :

    hb = df[df[base_column].str.contains('[|]')]
    
    # create two copies of the hybrids dataframe
    df1 = hb.copy()
    df2 = hb.copy()

    # apply split function to each column of each dataframe copy
    for c in split_columns:
        df1[c] = df1[c].apply(lambda x: x.split("|")[0])
        try:
            df2[c] = df2[c].apply(lambda x: x.split("|")[1])
        except IndexError:
            pass
    
    # append the two dataframes
    new_rows = df1.append(df2)

    # drop each hybrid row from the original dataframe
    df.drop(hb.index, inplace=True)

    # append new_rows to dataframe
    df = df.append(new_rows, ignore_index=True)

    return df

in order to fix the hyprid in all the columns we have (considering that we have multible hyprid in multible columns), we will have to repeat the function for each column everytime, and that will create copies everytime (identical but differnt in that column only), and we will base our analysis on it.
